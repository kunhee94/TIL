def all_list_sum(a):
    k=[]
    total=0
    for i in a:
        k+=i
    for j in k:
        total+=j         #이런식으로 아예 2중for문이 아니라 걍 밖으로 끄집어 내버리면 문제는 해결되                            긴함
    return total
