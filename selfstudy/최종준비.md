연도 기준으로 영화 가져오기

```python
import requests
from pprint import pprint


def get_year_list():
    base_url = 'https://api.themoviedb.org/3'
    path = '/discover/movie'

    params = {
        'api_key': '',
        'language': 'ko',
        'primary_release_year': 2000,
    }
    res = requests.get(base_url + path, params=params)
    year_list = res.json()
    return year_list


```
